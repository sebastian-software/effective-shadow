Quick Start: Presets

Ready-to-use shadows with zero configuration.

CSS Classes

import "@effective/shadow/shadows.css"

<div class="shadow-3">Card</div>

CSS Modules

import s from "@effective/shadow/shadows.module.css"

<div className={s["shadow-3"]}>Card</div>

Tailwind Plugin

// tailwind.config.js
import shadow from "@effective/shadow/tailwind"
export default { plugins: [shadow] }

<div class="shadow-effective-3">Card</div>

Quick Start: API

Full control with the JavaScript/TypeScript API.

Pre-generated Shadows

import { boxShadow, dropShadow } from "@effective/shadow"

// Apply directly
element.style.boxShadow = boxShadow[3]
element.style.filter = dropShadow[3]

Custom Shadows

import { buildShadow, toBoxShadow } from "@effective/shadow"

const shadow = buildShadow({
  shadowLayers: 4,
  finalOffsetY: 12,
  finalBlur: 24,
  finalAlpha: 0.25
})

const css = toBoxShadow(shadow)

Use Presets

import { generateCSS, effectivePreset } from "@effective/shadow"

// Generate CSS from any preset
const css = generateCSS(effectivePreset)

// Or create your own preset
const myPreset = { ...effectivePreset, elevations: [...] }
